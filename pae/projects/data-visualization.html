  4. Load the data

    Create a new cell and run:

    import pandas as pd
import matplotlib.pyplot as plt

# Load the data
df = pd.read_csv('population.csv')

# Look at the first few rows
print(df.head())
print(f"\nColumns: {df.columns.tolist()}")
print(f"Shape: {df.shape}")
  5. Get the top 10 countries by population

    Run this code:

    # The dataset has years as columns (1960, 1961, ... 2021)
# Let's use the most recent year: 2021
df_2021 = df[['Country Name', '2021']].copy()

# Rename the '2021' column to 'Value' for easier use
df_2021 = df_2021.rename(columns={'2021': 'Value'})

# Remove rows with missing values
df_2021 = df_2021.dropna()

# Get top 10 countries by population
top10 = df_2021.nlargest(10, 'Value')

# Display the top 10
print("Top 10 Countries by Population (2021):")
print(top10)
  6. Create a basic bar chart

    Run this code:

    # Create bar chart
plt.figure(figsize=(10, 6))
plt.bar(top10['Country Name'], top10['Value'])
plt.xticks(rotation=45, ha='right')
plt.title('Top 10 Countries by Population (2021)')
plt.xlabel('Country')
plt.ylabel('Population')
plt.tight_layout()
plt.show()
  7. Make it beautiful (REQUIRED for full points!)

    Run this customized version:

    # Create a beautiful chart
plt.figure(figsize=(12, 7), facecolor='lightblue')

# Create bars with custom colors
bars = plt.bar(top10['Country Name'], top10['Value'],
               color='darkblue',
               edgecolor='navy',
               linewidth=2)

# Add value labels on top of bars
for bar in bars:
    height = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2., height,
             f'{height/1e9:.1f}B',
             ha='center', va='bottom', fontsize=10)

# Customize the chart
plt.title('Top 10 Most Populous Countries (2021)', fontsize=18, fontweight='bold', color='darkblue')
plt.xlabel('Country', fontsize=14, fontweight='bold')
plt.ylabel('Population', fontsize=14, fontweight='bold')
plt.xticks(rotation=45, ha='right')

# Add grid for better readability
plt.grid(axis='y', alpha=0.3, linestyle='--')

# Format y-axis to show billions
ax = plt.gca()
ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda x, p: f'{x/1e9:.1f}B'))

plt.tight_layout()
plt.show()
  8. Optional: Compare countries over time

    Try this line chart:

    # Compare USA and China over time
countries = ['United States', 'China']
plt.figure(figsize=(12, 6))

for country in countries:
    # Get the row for this country
    country_row = df[df['Country Name'] == country].iloc[0]

    # Extract years and populations
    years = []
    populations = []

    # Loop through year columns (they're strings like '1960', '1961', etc.)
    for col in df.columns:
        if col.isdigit() and int(col) >= 1960:  # Only use year columns
            if pd.notna(country_row[col]):  # Skip missing values
                years.append(int(col))
                populations.append(country_row[col])

    # Plot the line
    plt.plot(years, populations, marker='o', label=country, linewidth=2, markersize=3)

plt.title('Population Growth: USA vs China', fontsize=16, fontweight='bold')
plt.xlabel('Year', fontsize=12)
plt.ylabel('Population', fontsize=12)
plt.legend()
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.show()
