Sample Python Code

# Move the robot in a square pattern
from vex import *

def draw_square():
    for i in range(4):
        drivetrain.drive_for(FORWARD, 200, MM)
        drivetrain.turn_for(RIGHT, 90, DEGREES)

# Start the program
pen.move(DOWN)
draw_square()
